#include "ParticleSet.h"


ParticleSet::ParticleSet(xml_node<> *node)
{
   using namespace rapidxml;
   using namespace std;
   
   //Get the name of the particle set and store it
   _name=CatchXmlNodeName(node);
   
   //Print some information
   cout << "Load information about the particle set " << _name << " ..." << endl;
   
   // Get NPart
   CatchXmlSubNodeValue(node,"NPart",_n_part);
   cout << "NPart = " << _n_part << "; ";
   // Get NPart
   CatchXmlSubNodeValue(node,"NHalfSpin",_n_half_spin);
   cout << "NHalfSpin = " << _n_half_spin << "; ";
   
}
